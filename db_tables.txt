equipment_categories
カテゴリIDとカテゴリ名
create table equipment_categories(
  id integer primary key,
  name text
);

equipments
備品ID、名前、カテゴリID（外部キー）
create table equipments(
  id integer primary key,
  name text,
  category integer,
  foreign key (category) references equipment_categories(id)
);

equipment_reservation_schedules
// id equipment_id project_id start_date end_date
備品予約ID、予約されてる備品ID（外部キー）、開始日、終了日
create table equipment_reservation_schedules(
  id integer primary key,
  equipment_id integer,
  start_date text,
  end_date text,
  foreign key (equipment_id) references equipments(id)
);

facilities
施設ID、施設名
create table facilities(
  id integer primary key,
  name text
);

facility_reservation_schedules
予約ID、施設ID（外部キー）、開始時刻、終了時刻
create table facility_reservation_schedules(
  id integer primary key,
  facility_id integer,
  start_date text,
  end_date text,
  foreign key (facility_id) references facilities(id)
);

user
id name password_digest evaluation role create_at update_at
create table users(
  id integer primary key,
  name text,
  email text,
  evaluation integer,
  role integer,
  create_at text,
  update_at text,
  foreign key (role) references roles(id)
);
